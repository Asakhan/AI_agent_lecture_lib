def is_complete(self) -> bool:
    """ëª¨ë“  íƒœìŠ¤í¬ ì™„ë£Œ ì—¬ë¶€ í™•ì¸"""
    if not self.tasks:
        return True
    
    for task in self.tasks:

    return True
